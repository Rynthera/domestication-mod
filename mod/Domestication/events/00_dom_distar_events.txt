				}
				is_domesticator_or_protectorate = no
			}
		}
		name = "distar.172.a"
		owner = {
			while = {
				count = 3
				random_owned_pop = {
					limit = {
						is_sapient = yes

							NOR = {
								has_trait = trait_cybernetic
								has_trait = trait_mechanical
								has_trait = trait_machine_unit
								is_species_class = ROBOT
							}
						}
					}
					modify_species = {
						species = this
						add_trait = trait_brainslug
						add_traits_at_start_of_list = yes
						effect = {
							set_citizenship_type = {
								country = root.owner
								type = citizenship_full
								cooldown = no
							}
						}
					}
				}
			}
			hidden_effect = {
				while = {
					count = 2
					random_owned_leader = {
						LIMIT = {
							leader_class = scientist
							species = {
								NOR = {
									has_trait = trait_cybernetic
									has_trait = trait_mechanical
									has_trait = trait_machine_unit
									is_species_class = ROBOT
								}
							}
						}
						add_trait = leader_trait_scientist_brainslug
					}
				}
				random_owned_leader = {
					LIMIT = {
						leader_class = governor
						species = {
							NOR = {
								has_trait = trait_cybernetic
								has_trait = trait_mechanical
								has_trait = trait_machine_unit
								is_species_class = ROBOT
							}
						}
					}
					add_trait = leader_trait_governor_brainslug
				}
				country_event = { id = distar.174 } #Pool leaders get trait
				country_event = { id = distar.175 } #Unrest is generated
			}
		}
	}

	option = {
		name = "distar.172.domestication"
		trigger = {
			owner = { is_domesticator_or_protectorate = yes }
		}
		icon = {
			icon = GFX_domesticators_icon
			text = dialog_domesticator
		}
		hidden_effect = {
			if = {
				limit = {
					leader = { can_obtain_new_floret = yes }
				}
				leader = { save_event_target_as = new_owner }
				owner = {
					country_event = { id = domestication_leader.20 days = 15 }
				}
			}
		}
		owner = {
			while = {
				count = 3
				random_owned_pop = {
					limit = {
						is_sapient = yes
						species = {
							NOR = {
								has_trait = trait_cybernetic
								has_trait = trait_mechanical
								has_trait = trait_machine_unit
								is_species_class = ROBOT
								is_same_species = root.owner.owner_species
								is_subspecies = root.owner.owner_species
							}
						}
					}
					modify_species = {
						species = this
						add_trait = trait_brainslug
						add_traits_at_start_of_list = yes
						effect = {
							set_citizenship_type = {
								country = root.owner
								type = citizenship_limited
								cooldown = no
							}
						}
					}
				}
			}
			if = {
				limit = {
					has_tech_option = tech_haustoric_implant
				}
				add_tech_progress = {
					tech = tech_haustoric_implant
					progress = 0.5
				}
			}
			else_if = {
				limit = {
					NOT = { has_technology = tech_haustoric_implant }
				}
				add_research_option = tech_haustoric_implant
				add_tech_progress = {
					tech = tech_haustoric_implant
					progress = 0.1
				}
			}
			else = {
				add_monthly_resource_mult = {
					resource = society_research
					value = @tier3researchreward
					min = @tier3researchmin
					max = @tier3researchmax
				}
			}
		}
	}

	option = {
		trigger = { # at least one sluggable pop in xenophobe empire
			hidden_trigger = {

					OR = {
						has_ethic = ethic_xenophobe
						has_ethic = ethic_fanatic_xenophobe

						NOR = {
							has_ethic = ethic_xenophobe
							has_ethic = ethic_fanatic_xenophobe
						}
					}
				}
			}
		}
		name = "distar.172.a"
		owner = {
			while = { # slug *up to* three non-phobe pops
				count = 3
				random_owned_pop = {
					LIMIT = {
						NOR = {
							has_ethic = ethic_xenophobe
							has_ethic = ethic_fanatic_xenophobe

							NOR = {
								has_trait = trait_cybernetic
								has_trait = trait_mechanical
								has_trait = trait_machine_unit

							}
						}
						is_sapient = yes
					}
					modify_species = {
						species = this
						add_trait = trait_brainslug
						add_traits_at_start_of_list = yes
						effect = {
							set_citizenship_type = {
								country = root.owner
								type = citizenship_full
								cooldown = no
							}
						}
					}
				}
			}
			hidden_effect = {
				country_event = { id = distar.174 } #Pool leaders get trait
				country_event = { id = distar.175 } #Unrest is generated
			}
		}

	option = {
		name = "distar.172.b"

	}
