population_control_yes = {
	can_reproduce = no
	pop_modifier = {
		pop_happiness = -0.1
	}

	triggered_pop_modifier = {
		potential = {
			has_haustorics = yes
		}
		pop_happiness = 0.1
	}

	allow = {
		hidden_trigger = { exists = from }
		from = {
			NAND = {
				has_policy_flag = population_controls_not_allowed
				is_domesticator_or_protectorate = no
			}
		}
		if = {
			limit = {
				OR = {
					has_citizenship_type = { country = from type = citizenship_purge }
					has_citizenship_type = { country = from type = citizenship_purge_machine }
					has_haustorics = yes
				}
			}
			hidden_trigger = { always = yes }
		}
		else = {
			custom_tooltip = {
				fail_text = same_species_disallowed
				NOT = { is_same_species = from }
			}
			custom_tooltip = {
				fail_text = is_broken_shackles_origin
				NOT = {
					from = {
						has_origin = origin_broken_shackles
					}
				}
			}
			custom_tooltip = {
				fail_text = empire_xenophile
				from = {
					NOR = {
						has_ethic = ethic_xenophile
						has_ethic = ethic_fanatic_xenophile
					}
				}
			}
			custom_tooltip = {
				fail_text = DOMESTICATOR_POP_CONTROLS_NOT_ALLOWED
				from = {
					NOT = {
						has_policy_flag = population_controls_not_allowed
					}
				}
			}
		}
	}
	ai_will_do = {
		factor = 2
		modifier = {
			factor = 0
			NOR = {
				has_citizenship_type = { country = from type = citizenship_purge_machine }
				has_citizenship_type = { country = from type = citizenship_purge }
			}
		}
	}
}

		}

			fail_text = SPECIES_STERILE_HAUSTORIC_IMPLANT
			has_haustorics = no
		}
	}
	ai_will_do = {
		factor = 1
	}
